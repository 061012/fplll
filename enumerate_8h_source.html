77  void enumerate(int first, int last, FT &fmaxdist, long fmaxdistexpo,
78  const vector<FT> &target_coord = vector<FT>(),
79  const vector<enumxt> &subtree = vector<enumxt>(),
80  const vector<enumf> &pruning = vector<enumf>(), bool dual = false,
81  bool subtree_reset = false)
82  {
83  // check for external enumerator and use that
84  if (get_external_enumerator() != nullptr && subtree.empty() && target_coord.empty())
85  {
86  if (enumext.get() == nullptr)
87  enumext.reset(new ExternalEnumeration<FT>(_gso, _evaluator));
88  if (enumext->enumerate(first, last, fmaxdist, fmaxdistexpo, pruning, dual))
89  {
90  _nodes = enumext->get_nodes();
91  return;
92  }
93  }
94  // if external enumerator is not available, not possible or when it fails then fall through to
95  // fplll enumeration
96  if (enumdyn.get() == nullptr)
97  enumdyn.reset(new EnumerationDyn<FT>(_gso, _evaluator, _max_indices));
98  enumdyn->enumerate(first, last, fmaxdist, fmaxdistexpo, target_coord, subtree, pruning, dual,
99  subtree_reset);
100  _nodes = enumdyn->get_nodes();
101  }
102 
103  inline uint64_t get_nodes() const { return _nodes; }
104 
105 private:
106  MatGSO<Integer, FT> &_gso;
107  Evaluator<FT> &_evaluator;
108  vector<int> _max_indices;
109  std::unique_ptr<EnumerationDyn<FT>> enumdyn;
110  std::unique_ptr<ExternalEnumeration<FT>> enumext;
111  uint64_t _nodes;
