37 template<class T>
38 void rotate_by_swap(vector<T>& v, int first, int middle, int last) {
39  // Algorithm from STL code
40  reverse_by_swap(v, first, middle - 1);
41  reverse_by_swap(v, middle, last);
42  for (; first < middle && middle <= last; first++, last--) {
43  v[first].swap(v[last]);
44  }
45  if (first == middle)
46  reverse_by_swap(v, middle, last);
47  else
48  reverse_by_swap(v, first, middle - 1);
49 }
