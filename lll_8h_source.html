72 template <class ZT, class FT>
73 inline bool LLLReduction<ZT, FT>::size_reduction(int kappa_min, int kappa_end)
74 {
75  if (kappa_end == -1)
76  kappa_end = m.d;
77  for (int k = kappa_min; k < kappa_end; k++)
78  {
79  if ((k > 0 && !babai(k, k)) || !m.update_gso_row(k))
80  return false;
81  }
82  return set_status(RED_SUCCESS);
83 }

85 template <class ZT, class FT> inline bool LLLReduction<ZT, FT>::early_reduction(int start)
86 {
87  m.lock_cols();
88  if (verbose)
89  {
90  cerr << "Early reduction start=" << start + 1 << endl;
91  }
92  for (int i = start; i < m.d; i++)
93  {
94  if (!babai(i, start))
95  return false;
96  }
97  m.unlock_cols();
98  last_early_red = start;
99  return true;
100 }
