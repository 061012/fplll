511 template <class ZT, class FT> inline FT &MatGSO<ZT, FT>::get_gram(FT &f, int i, int j)
512 {
513  FPLLL_DEBUG_CHECK(i >= 0 && i < n_known_rows && j >= 0 && j <= i && j < n_source_rows &&
514  !in_row_op_range(i));
515  if (enable_int_gram)
516  f.set_z(g(i, j));
517  else
518  {
519  if (gf(i, j).is_nan())
520  {
521  dot_product(gf(i, j), bf[i], bf[j], n_known_cols);
522  }
523  f = gf(i, j);
524  }
525  return f;
526 }

607 template <class ZT, class FT> inline void MatGSO<ZT, FT>::create_rows(int n_new_rows)
608 {
609  FPLLL_DEBUG_CHECK(!cols_locked);
610  int old_d = d;
611  d += n_new_rows;
612  b.set_rows(d);
613  for (int i = old_d; i < d; i++)
614  {
615  for (int j = 0; j < b.get_cols(); j++)
616  {
617  b[i][j] = 0;
618  }
619  }
620  if (enable_transform)
621  {
622  u.set_rows(d);
623  for (int i = old_d; i < d; i++)
624  for (int j = 0; j < u.get_cols(); j++)
625  u[i][j] = 0;
626  }
627  size_increased();
628  if (n_known_rows == old_d)
629  discover_all_rows();
630 }
