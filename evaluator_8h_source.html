47 template <class FT> class Evaluator
48 {
49 public:
50  Evaluator(size_t nr_solutions = 1,
51  EvaluatorStrategy update_strategy = EVALSTRATEGY_BEST_N_SOLUTIONS,
52  bool find_subsolutions = false)
53  : max_sols(nr_solutions), strategy(update_strategy), findsubsols(find_subsolutions),
54  sol_count(0)
55  {
56  FPLLL_CHECK(nr_solutions > 0, "Evaluator: nr_solutions must be strictly positive!");
57  FPLLL_CHECK(strategy <= 2, "Evaluator: invalid strategy");
58  }
59  virtual ~Evaluator() {}
60 
62  size_t max_sols;
63  EvaluatorStrategy strategy;
64  bool findsubsols;
65 
67  // multimap storing solutions mapped by their length
68  // the longest solution is the first, the shortest solution last
69  typedef std::multimap<FT, std::vector<FT>, std::greater<FT>> container_t;
70  container_t solutions;
71  size_t sol_count;
72 
74  std::vector<std::pair<FT, std::vector<FT>>> sub_solutions;
75 
77  typename container_t::const_reverse_iterator begin() const { return solutions.rbegin(); }
78  typename container_t::const_reverse_iterator end() const { return solutions.rend(); }
79  typename container_t::reverse_iterator begin() { return solutions.rbegin(); }
80  typename container_t::reverse_iterator end() { return solutions.rend(); }
81  size_t size() const { return solutions.size(); }
82  bool empty() const { return solutions.empty(); }
83 
85  virtual void eval_sol(const vector<FT> &new_sol_coord, const enumf &new_partial_dist,
86  enumf &max_dist) = 0;
87 
88  virtual void eval_sub_sol(int offset, const vector<FT> &new_sub_sol_coord,
89  const enumf &sub_dist) = 0;
90 
91  virtual void set_normexp(long norm_exp) { normExp = norm_exp; }
92  long normExp;
93 
94 protected:
96  virtual enumf calc_enum_bound(const FT &dist) const
97  {
98  FT tmp;
99  tmp.mul_2si(dist, -normExp);
100  return tmp.get_d(GMP_RNDU);
101  }
102 
104  void process_sol(const FT &dist, const vector<FT> &coord, enumf &max_dist)
105  {
106  ++sol_count;
107  solutions.emplace(dist, coord);
108  switch (strategy)
109  {
110  case EVALSTRATEGY_BEST_N_SOLUTIONS:
111  if (solutions.size() < max_sols)
112  return;
113  // remove the longest solution, and use the new longest dist to update max_dist
114  if (solutions.size() > max_sols)
115  solutions.erase(solutions.begin());
116  max_dist = calc_enum_bound(solutions.begin()->first);
117  break;
118 
119  case EVALSTRATEGY_OPPORTUNISTIC_N_SOLUTIONS:
120  // always use dist to update max_dist
121  max_dist = calc_enum_bound(dist);
122  if (solutions.size() <= max_sols)
123  return;
124  // remove longest solution
125  solutions.erase(solutions.begin());
126  break;
127 
128  case EVALSTRATEGY_FIRST_N_SOLUTIONS:
129  if (solutions.size() < max_sols)
130  return;
131  // when desired nr of solutions has been reached, set enum bound to zero
132  max_dist = 0;
133  break;
134 
135  default:
136  FPLLL_CHECK(false, "Evaluator: invalid strategy switch!");
137  }
138  }
139 };

146 template <class FT> class FastEvaluator : public Evaluator<FT>
147 {
148 public:
149  using Evaluator<FT>::max_sols;
150  using Evaluator<FT>::strategy;
151  using Evaluator<FT>::findsubsols;
152  using Evaluator<FT>::normExp;
153  using Evaluator<FT>::sub_solutions;
154 
155  FastEvaluator(size_t nr_solutions = 1,
156  EvaluatorStrategy update_strategy = EVALSTRATEGY_BEST_N_SOLUTIONS,
157  bool find_subsolutions = false)
158  : Evaluator<FT>(nr_solutions, update_strategy, find_subsolutions)
159  {
160  }
161  virtual ~FastEvaluator() {}
162 
163  virtual void eval_sol(const vector<FT> &new_sol_coord, const enumf &new_partial_dist,
164  enumf &max_dist)
165  {
166  FT dist = new_partial_dist;
167  dist.mul_2si(dist, normExp);
168 
169  // store solution and update max_dist according to strategy
170  this->process_sol(dist, new_sol_coord, max_dist);
171  }
172 
173  virtual void eval_sub_sol(int offset, const vector<FT> &new_sub_sol_coord, const enumf &sub_dist)
174  {
175  FT dist = sub_dist;
176  dist.mul_2si(dist, normExp);
177 
178  sub_solutions.resize(std::max(sub_solutions.size(), std::size_t(offset + 1)));
179 
180  if (sub_solutions[offset].second.empty() || dist < sub_solutions[offset].first)
181  {
182  sub_solutions[offset].first = dist;
183  sub_solutions[offset].second = new_sub_sol_coord;
184  for (int i = 0; i < offset; ++i)
185  sub_solutions[offset].second[i] = 0.0;
186  }
187  }
188 };
