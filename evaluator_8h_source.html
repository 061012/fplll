112  virtual void eval_sol(const vector<FT> &new_sol_coord, const enumf &new_partial_dist,
113  enumf &max_dist)
114  {
115  if (max_aux_sols != 0 && !sol_coord.empty())
116  {
117  aux_sol_coord.emplace_front(std::move(sol_coord));
118  aux_sol_dist.emplace_front(sol_dist);
119  if (aux_sol_coord.size() > max_aux_sols)
120  {
121  aux_sol_coord.pop_back();
122  aux_sol_dist.pop_back();
123  }
124  }
125  sol_coord = new_sol_coord;
126  max_dist = sol_dist = new_partial_dist;
127  new_sol_flag = true;
128  }
129 
130  virtual void eval_sub_sol(int offset, const vector<FT> &new_sub_sol_coord, const enumf &sub_dist)
131  {
132  sub_sol_coord.resize(std::max(sub_sol_coord.size(), std::size_t(offset + 1)));
133  sub_sol_dist.resize(sub_sol_coord.size(), -1.0);
134  if (sub_sol_dist[offset] == -1.0 || sub_dist < sub_sol_dist[offset])
135  {
136  sub_sol_coord[offset] = new_sub_sol_coord;
137  for (int i = 0; i < offset; ++i)
138  sub_sol_coord[offset][i] = 0.0;
139  sub_sol_dist[offset] = sub_dist;
140  }
141  }
