55  {
56 
57  Pruning pruning = Pruning();
58  int start_descent = block_size - level;
59 
60  if (start_descent > block_size)
61  start_descent = block_size;
62 
63  if (start_descent < 1)
64  start_descent = 1;
65 
66  pruning.coefficients.resize(block_size);
67  for (int k = 0; k < start_descent; k++)

70  }
